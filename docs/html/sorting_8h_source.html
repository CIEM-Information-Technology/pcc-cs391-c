101{

103 int n1 = m - l + 1;

106 /* create temp arrays */

110 for (i = 0; i < n1; i++)
111 L[i] = arr[l + i];
112 for (j = 0; j < n2; j++)
113 R[j] = arr[m + 1 + j];
114

117 j = 0; // Initial index of second subarray

120 if (L[i] <= R[j]) {
121 arr[k] = L[i];
122 i++;
123 }

126 j++;

177void heapify(int a[], int n, int i)
178{
179 int largest = i;
180 int left = 2 * i + 1;
181 int right = 2 * i + 2;
182
183 if(left < n && a[left] > a[largest])
184 {
185 largest = left;
186 }
187 if(right < n && a[right] > a[largest])
188 {
189 largest = right;
190 }
191 if(largest != i)
192 {
193 swap(&a[i], &a[largest]);
194 heapify(a, n, largest);
195 }
196}
